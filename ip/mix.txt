void lower_case(char arr[], int len){
    for(int pos = 0; pos<len;pos++){
        if ((arr[pos] >= 'A') && (arr[pos] <='Z')){
            printf("%c", arr[pos] + 32);
            continue;
        }
        printf("%c", arr[pos]);
    }
}
